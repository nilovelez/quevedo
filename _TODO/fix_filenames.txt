function quevedo_file_name(){
	/**
	 * https://developer.wordpress.org/reference/functions/remove_accents/
	 */
	function quevedo_sanitize_file_name_filter ( $filename, $filename_raw ) {
	    return remove_accents( $filename );
	}
	add_filter( 'sanitize_file_name', 'quevedo_sanitize_file_name_filter', 10, 2 );
}

add_action( 'init', 'quevedo_file_name' );
